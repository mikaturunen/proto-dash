/** 
 * Initializes the express server itself.

 */ 
var initServer = () => {
    var deferred = Q.defer();
    
    var server = app.listen(app.get("port"), () => {
        console.log("Node app is running at localhost:" + app.get('port'));
        deferred.resolve(server);
    });
    
    return deferred.promise;
};
